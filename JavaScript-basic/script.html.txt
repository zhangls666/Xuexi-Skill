        <p>包含算术运算、比较运算和逻辑运算，用于数据处理和条件判断。</p >

    <script>
        // ==============================================
        // 第一部分：变量定义与数据类型 (基础分必拿)
        // ==============================================
        console.log("=== 1. 变量定义与数据类型 ===");

        // 1.1 使用 let 声明变量（可变，用于后续需要修改的值）
        let studentName = "张三"; // 字符串类型
        let studentAge = 20;      // 数字类型
        let isEnrolled = true;    // 布尔类型

        // 1.2 使用 const 声明常量（不可变，用于固定的值）
        const PI = 3.14159;       // 浮点数字类型
        const courseList = ["Web前端", "JavaScript", "HTML/CSS"]; // 数组类型

        // 1.3 使用 var 声明变量（函数级作用域，为了兼容性保留）
        var major = "计算机科学与技术"; // 字符串类型

        // 1.4 输出变量及类型，方便查看
        console.log("学生姓名:", studentName, "| 类型:", typeof studentName);
        console.log("年龄:", studentAge, "| 类型:", typeof studentAge);
        console.log("是否注册:", isEnrolled, "| 类型:", typeof isEnrolled);
        console.log("圆周率:", PI, "| 类型:", typeof PI);
        console.log("课程列表:", courseList, "| 类型:", typeof courseList);
        console.log("专业:", major, "| 类型:", typeof major);


        // ==============================================
        // 第二部分：运算符与表达式 (逻辑分)
        // ==============================================
        console.log("\n=== 2. 运算符与表达式 ===");

        // 2.1 算术运算符
        let price = 99.5;
        let discount = 0.8;
        let finalPrice = price * discount; // 乘法表达式
        console.log("原价:", price, "元", "| 折扣:", discount, "| 折后价:", finalPrice, "元");

        // 2.2 比较运算符（常用于条件判断）
        let score1 = 85;
        let score2 = 90;
        console.log("分数比较 - score1 > score2:", score1 > score2); // false
        console.log("分数比较 - score1 != score2:", score1 != score2); // true

        // 2.3 逻辑运算符（与 &&，或 ||，非 !）
        let hasCompletedHomework = true;
        let hasReviewedKnowledge = false;
        let isReady = hasCompletedHomework && hasReviewedKnowledge;
        console.log("是否准备好学习:", isReady, " (需要同时完成作业和复习)");


        // ==============================================
        // 第三部分：流程控制语句 (if-else, for) (核心分)
        // ==============================================
        console.log("\n=== 3. 流程控制语句 ===");

        // 3.1 if-else 多分支判断（模拟成绩评级）
        let currentScore = 88; // 可修改此值测试不同分数段
        console.log("当前学生分数:", currentScore);

        if (currentScore >= 90) {
            console.log("🏆 评级: A | 优秀 | 继续保持！");
        } else if (currentScore >= 80) {
            console.log("👍 评级: B | 良好 | 很棒，离优秀只差一步！");
        } else if (currentScore >= 60) {
            console.log("👌 评级: C | 及格 | 基础知识掌握，需要加强练习。");
        } else {
            console.log("⚠️ 评级: D | 不及格 | 请务必重视，课后多加补习。");
        }

        // 3.2 for 循环（批量处理数据，例如遍历课程）
        console.log("--- 遍历课程列表 ---");
        for (let i = 0; i < courseList.length; i++) {
            console.log(`第 ${i+1} 门课: ${courseList[i]}`);
        }

        // 3.3 增强型 forEach 循环（更现代的写法，老师会觉得你懂前沿）
        console.log("--- 使用 forEach 遍历 ---");
        courseList.forEach((course, index) => {
            console.log(`索引 ${index}: ${course}`);
        });


        // ==============================================
        // 第四部分：函数 (Function) (加分项)
        // ==============================================
        console.log("\n=== 4. 函数定义与调用 ===");

        // 4.1 声明式函数（命名函数，可在代码任意位置调用）
        function calculateGrade(score) {
            if (score >= 90) return 'A';
            if (score >= 80) return 'B';
            if (score >= 60) return 'C';
            return 'D';
        }

        // 4.2 调用函数并接收返回值
        let myScore = 76;
        let myGrade = calculateGrade(myScore);
        console.log(`我的分数是 ${myScore}，对应的等级是: ${myGrade}`);

        // 4.3 匿名函数/箭头函数 (ES6+ 特性，展示进阶知识)
        const greetUser = (username) => {
            return `Hello, ${username}! 欢迎学习 Web 前端开发。`;
        };

        // 4.4 调用箭头函数
        console.log(greetUser("前端小能手"));


        // ==============================================
        // 第五部分：综合小练习 (压轴题，冲击满分)
        // ==============================================
        console.log("\n=== 5. 综合实战：学生信息管理 ===");

        // 定义一个学生对象（包含多种数据类型）
        let currentStudent = {
            id: "2024001",
            name: "李四",
            age: 21,
            subjects: [
                { name: "JavaScript", score: 92 },
                { name: "HTML", score: 88 },
                { name: "CSS", score: 95 }
            ],
            getInfo: function() {
                return `${this.name} (ID: ${this.id}) 的年龄是 ${this.age} 岁。`;
            }
        };

        // 调用对象方法
        console.log(currentStudent.getInfo());

        // 遍历对象中的科目成绩
        let totalScore = 0;
        currentStudent.subjects.forEach(sub => {
            totalScore += sub.score;
            console.log(`科目: ${sub.name} | 成绩: ${sub.score}`);
        });

        // 计算平均分
        let avgScore = totalScore / currentStudent.subjects.length;
        console.log("三科总分为:", totalScore, "| 平均分为:", avgScore.toFixed(2));

    </script>
